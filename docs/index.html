Quick Start

Using Docker

# Create web network
docker network create web

# Run NubDB
docker run -d \
  --name nubdb-server \
  --network web \
  -p 6379:6379 \
  -v nubdb-data:/data \
  nubdb:latest

# Test connection
echo "SET hello world" | nc localhost 6379

Using Docker Compose

# Create network
docker network create web

# Start NubDB
docker-compose up -d

# View logs
docker-compose logs -f

# Test connection
echo "GET hello" | nc localhost 6379

Build from Source

# Clone repository
git clone https://github.com/nub-coders/nubdt.git
cd nubdt

# Build
make build

# Run in server mode
make server

# Or use zig directly
zig build -Doptimize=ReleaseFast
./zig-out/bin/nubdt --server

Deploy to Kubernetes

# Apply manifests
kubectl apply -f k8s/namespace.yaml
kubectl apply -f k8s/deployment.yaml

# Check status
kubectl get pods -n nubdb

# Port forward
kubectl port-forward -n nubdb svc/nubdb-service 6379:6379

API Reference

SET

SET key value [ttl]

Store a key-value pair with optional TTL (seconds)

SET mykey "Hello, World!"
SET counter 100 60  # Expires in 60 seconds
Response: OK

GET

GET key

Retrieve value by key

GET mykey
Response: "Hello, World!" or (nil)

DELETE

DELETE key

Remove a key from the database

DELETE mykey
Response: OK or (not found)

EXISTS

EXISTS key

Check if key exists

EXISTS mykey
Response: 1 (exists) or 0 (not found)

INCR

INCR key

Increment integer value by 1

INCR counter
Response: New value (integer)

DECR

DECR key

Decrement integer value by 1

DECR counter
Response: New value (integer)

SIZE

SIZE

Get number of keys in database

SIZE
Response: N keys

CLEAR

CLEAR

Delete all keys from database

CLEAR
Response: OK

Code Examples

Python Client

import socket

class NubDBClient:
    def __init__(self, host='nubdb-server', port=6379):
        self.sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.sock.connect((host, port))
    
    def set(self, key, value):
        cmd = f"SET {key} {value}\n"
        self.sock.sendall(cmd.encode())
        return self.sock.recv(1024).decode().strip()
    
    def get(self, key):
        cmd = f"GET {key}\n"
        self.sock.sendall(cmd.encode())
        return self.sock.recv(1024).decode().strip()

# Usage
client = NubDBClient()
client.set('greeting', 'Hello from Python!')
result = client.get('greeting')
print(result)  # "Hello from Python!"

Node.js Client

const net = require('net');

class NubDBClient {
    constructor(host = 'nubdb-server', port = 6379) {
        this.client = net.createConnection({ host, port });
    }

    set(key, value) {
        return new Promise((resolve) => {
            this.client.write(`SET ${key} ${value}\n`);
            this.client.once('data', (data) => {
                resolve(data.toString().trim());
            });
        });
    }

    get(key) {
        return new Promise((resolve) => {
            this.client.write(`GET ${key}\n`);
            this.client.once('data', (data) => {
                resolve(data.toString().trim());
            });
        });
    }
}

// Usage
const client = new NubDBClient();
await client.set('greeting', 'Hello from Node.js!');
const result = await client.get('greeting');
console.log(result);  // "Hello from Node.js!"

Go Client

package main

import (
    "bufio"
    "fmt"
    "net"
)

type NubDBClient struct {
    conn net.Conn
}

func NewClient(host string, port string) (*NubDBClient, error) {
    conn, err := net.Dial("tcp", host+":"+port)
    if err != nil {
        return nil, err
    }
    return &NubDBClient{conn: conn}, nil
}

func (c *NubDBClient) Set(key, value string) (string, error) {
    cmd := fmt.Sprintf("SET %s %s\n", key, value)
    _, err := c.conn.Write([]byte(cmd))
    if err != nil {
        return "", err
    }
    response, _ := bufio.NewReader(c.conn).ReadString('\n')
    return response, nil
}

func main() {
    client, _ := NewClient("nubdb-server", "6379")
    result, _ := client.Set("greeting", "Hello from Go!")
    fmt.Println(result)
}

Rust Client

use std::io::{Read, Write};
use std::net::TcpStream;

struct NubDBClient {
    stream: TcpStream,
}

impl NubDBClient {
    fn new(host: &str, port: u16) -> std::io::Result<Self> {
        let stream = TcpStream::connect((host, port))?;
        Ok(NubDBClient { stream })
    }

    fn set(&mut self, key: &str, value: &str) -> std::io::Result<String> {
        let cmd = format!("SET {} {}\n", key, value);
        self.stream.write_all(cmd.as_bytes())?;
        
        let mut response = String::new();
        self.stream.read_to_string(&mut response)?;
        Ok(response.trim().to_string())
    }
}

fn main() {
    let mut client = NubDBClient::new("nubdb-server", 6379).unwrap();
    let result = client.set("greeting", "Hello from Rust!").unwrap();
    println!("{}", result);
}
